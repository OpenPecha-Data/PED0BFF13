བསྡུ་སྒྲིག་གསལ་བཤད།

 དེ་ཡང་རྩ་རྒྱུད་སྒྱུ་འཕྲུལ་དྲྭ་བ་གསང་བའི་སྙིང་པོའི་འགྲེལ་པ་ཀུན་བཟང་ཐུགས་ཀྱི་ཏི་ཀ་ཞེས་བྱ་བ་འདི་ནི། རྒྱལ་བའི་སྲས་པོ་གཞན་ཕན་མཐའ་ཡས་འོད་ཟེར་མཆོག་གིས་ལྗགས་རྩོམ་གནང་བ་ཞིག་སྟེ། འདིའི་མ་དཔེ་ནི་སྔར་རྫོགས་ཆེན་ཤིང་པར་གྱི་བུ་དཔེ་ཞིག་རྩེ་པོ་ཏཱ་ལར་བཞུགས་པ་ཙམ་ལས། མདོ་དབུས་ཁམས་ཀུན་ཏུ་བཙལ་ཀྱང་དཔེ་གཞན་པ་མ་རྙེད། གཞུང་འདིའི་རྩ་བའི་ས་བཅད་གཉིས་ཏེ། སྤྱི་དོན་ལོ་རྒྱུས་ཀྱི་སྒོ་ནས་ཁོག་དབུབ་པ་དང༌། བྱེ་བྲག་གཞུང་དོན་གྱི་སྒོ་ནས་རྒྱས་པར་བཤད་པ་གཉིས་ལས། དང་པོ་བརྒྱུད་ཚུལ་དངོས་དང༌། དེ་ལ་ཀླན་ཀ་སྤངས་པ་གཉིས་ཡོད་པའི་དང་པོ་ལ། རྒྱལ་བ་དགོངས་རྒྱུད་དང༌། རིག་འཛིན་བརྡ་རྒྱུད། གང་ཟག་རྣར་བརྒྱུད་གསུམ་ཡོད་པའི་ནང་ནས་གང་ཟག་རྣར་བརྒྱུད་བཤད་པའི་སྐབས་སུ། རྒྱ་གར་འཕགས་པའི་ཡུལ་དུ་བརྒྱུད་ཚུལ་མདོར་བསྡུས་ཙམ་སོང་ནས། བོད་དུ་རྩ་རྒྱུད་གསང་བའི་སྙིང་པོ་སྔ་ཕྱིར་ལན་གསུམ་འགྱུར་བ་ལས། ཕྱི་འགྱུར་པཎ་ཆེན་དྲི་མེད་བཤེས་གཉེན་དང༌། ལོ་ཙཱ་བ་རྨ་གཉགས་གཉིས་ཀྱིས་འགྱུར་ཚུལ་འཆད་པའི་མཚམས་ནས་ཚང་མི་འདུག །འོན་ཀྱང་དཔེ་ཉིད་ལས་གསལ་བ་ལྟར་ན་ཁྱོན་ཤོག་གྲངས་བདུན་ཅུ་དོན་གསུམ་པ་དེ་ཉིད་ལ་བྱོན་ཞེས་བྲིས་པ་མ་ཟད། གཞུང་གི་ཚིག་མཚམས་ལའང་རྫོགས་ཚིག་བྲིས་པ་དང༌། མཚམས་ཐིག་གིས་བཅད་དེ་མཇུག་རྫོགས་པའི་རྣམ་པ་བྱས་འདུག་ན་ཡང༌། རྩ་གཞུང་དངོས་ཀྱི་ས་བཅད་ལ་གཞིག་ན་གོང་ཞུས་ལྟར། སྤྱི་དོན་ལོ་རྒྱུས་ཀྱི་སྒོ་ནས་ཁོག་དབུབ་པ་དང༌། བྱེ་བྲག་གཞུང་དོན་གྱི་སྒོ་ནས་རྩ་གཞུང་རྒྱས་པར་བཤད་པ་གཉིས་ཡོད་པའི་ནང་ནས། ས་བཅད་དང་པོའི་ནང་གསེས་གང་ཟག་རྣར་བརྒྱུད་ཀྱི་མཇུག་ཡང་མ་ཚང་བ་ཙམ་དུ་ཟད། ད་ལམ་དཔེ་འདི་ལས་མངོན་པ་ལྟར་ན། དང་པོ་ཤིང་པར་དུ་བརྐོ་སྐབས་ནས་མཇུག་མི་ཚང་བའི་རྣམ་པ་ཞིག་འདུག་པས། རྩོམ་པ་པོའི་བརྩམས་ཆོས་འཕྲོ་ཅན་དུ་ལུས་པའམ། ཡང་ན་ཤིང་པར་དུ་འགོད་སྐབས་དཔེ་ཆ་ཚང་ཞིག་མ་རྙེད་པ་གང་ཡིན་ད་ལམ་གསལ་བཤད་ཞུ་མ་ཤེས།

འདིའི་མ་དཔེ་ནི་རྩེ་པོ་ཏཱ་ལའི་དཔེ་མཛོད་ཁང་གི་རྙིང་མའི་གསུང་འབུམ་གྲལ། རྒྱལ་སྲས་གཞན་ཕན་མཐའ་ཡས་འོད་ཟེར་མཆོག་གི་གསུང་འབུམ་པོད་གཉིས་ཡོད་པའི་ནང་དུ་བཞུགས། དཔེ་འདི་སྔ་མོ་ནས་ང་ཚོར་རྙེད་སོན་བྱུང་ཡང་པར་རྙིང་ཡིན་པའི་རྐྱེན་གྱིས་ཡི་གེ་མི་གསལ་བ་མང་བ་དང༌། ཆ་མི་ཚང་བ་བཅས་ཀྱི་དཔེ་གཞན་པ་ཞིག་རྙེད་རེ་བྱས་ཏེ་ད་སྔོན་པར་སྐྲུན་ཞུ་རྒྱུ་མ་བྱུང༌། འོན་ཀྱང་ད་ལྟའི་ཆར་དཔེ་གཞན་པ་ཡང་རྙེད་པའི་རེ་བ་ཆེན་པོ་མེད་པ་དང༌། འདིའི་སྤྱི་དོན་ལོ་རྒྱུས་འཆད་པའི་ནང་གསེས་སུ། རྩ་རྒྱུད་གསང་བ་སྙིང་པོའི་བསྟན་དོན་རྒྱུད་ཀྱི་དངོས་པོ་བཅུ་ཆ་ཚང་ཞིག་བསྟན་ཡོད་པ་དང༌། དཔེ་རྒྱུན་མི་ནུབ་པ་སོགས་ཀྱི་ཆེད་དུ་དམིགས་ཏེ་པར་དུ་བསྐྲུན་པ་ཡིན། འདིར་མ་དཔེ་གསལ་པོ་མེད་པས་ཐག་མ་ཆོད་པ་འགའ་ཤས་ནི་གུག་རྟགས་ནང་དུ་གསལ་བཤད་ཞུས་ཏེ་བཞག་ཡོད། དེ་མིན་ནི་མ་དཔེ་སོར་བཞག་གིས་ཞུ་དག་བྱས་ཏེ་བཀྲམ་པ་ཡིན། འབྱུང་ཆར་མཁྱེན་ལྡན་ཀློག་པ་པོ་རྣམས་ཀྱིས་དཔེ་གཞན་པ་རྙེད་སོན་བྱུང་ན་དག་ཞུས་དང་འབྲེལ། མཇུག་མི་ཚང་བ་ཡང་ཁ་སྐོང་གནང་ཐུབ་མིན་སོགས་ཐུགས་བཞག་ཡོད་པ་མཁྱེན། མཁྱེན། ད་ལམ་ང་ཚོའི་མཁྱེན་བརྩེའི་འོད་སྣང་ཞེས་པའི་དཔེ་ཚོགས་འདི་སྤྱི་ཚོགས་ཐོག་འགྲེལ་སྤེལ་བྱེད་སྐབས། བོད་ལྗོངས་མི་དམངས་དཔེ་སྐྲུན་ཁང་གིས་མཉམ་ལས་ཀྱི་སྒོ་ནས། ཞུ་དག་དང༌། འཆར་འགོད་སོགས་རོགས་རམ་རྒྱ་ཆེན་གནང་བ་ལ་སྙིང་ཐག་པ་ནས་ཐུགས་རྗེ་ཆེ་ཞུ་བ་དང་བཅས། དཔལ་བརྩེགས་བོད་ཡིག་དཔེ་རྙིང་ཞིབ་འཇུག་ཁང་ནས་ཀརྨ་བདེ་ལེགས་ཀྱིས། 2008 ལོའི་ཟླ 6 ཚེས 10 ཉིན་ལ། བསྡུ་སྒྲིག་གསལ་བཤད།བསྡུ་སྒྲིག་གསལ་བཤད། -PAGE 2--PAGE 3--PAGE 1-
